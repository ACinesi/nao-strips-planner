Initial state:  At(A), Level(high), Hand(empty)
Goal state:     Bring(Ball)

Actions:
        // move from X to Y
        Move(X, Y)
        Preconditions:  At(X), !Level(low)
        Postconditions: !At(X), At(Y)
               
        // stand up
        StandUp(Location)
        Preconditions:  At(Location), Level(low)
        Postconditions: Level(high), !Level(low)
               
        // sit down 
        SitDown(Location)
        Preconditions:  At(Location), Level(high)
        Postconditions: Level(low), !Level(high)  
               
        // take the ball
        TakeBall(Location)
        Preconditions:  BallAt(Location), At(Location),Hand(empty), !Level(low)
        Postconditions: Hand(full), !Hand(empty), Bring(Ball)

        // give the ball
        GiveBall(Location)
        Preconditions:  Bring(Ball), Hand(full), !Level(low), At(Location)
        Postconditions: !Bring(Ball), !Hand(full), Hand(empty)

        // find a person
        FindPerson(Name,Location)
        Preconditions:  !Level(low)
        Postconditions: PersonAt(Name,Location)

        // find a ball
        FindBall(Location)
        Preconditions:  !Level(low),Hand(empty)
        Postconditions: BallAt(Location)
